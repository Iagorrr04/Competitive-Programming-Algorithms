Then 

\begin{equation}
  XOR(0, x) =

    x & \text{if } x \equiv 0 \pmod{4} \\
    1 & \text{if } x \equiv 1 \pmod{4} \\
    x + 1 & \text{if } x \equiv 2 \pmod{4} \\
    0 & \text{if } x \equiv 0 \pmod{4} \\
  \end{cases}
\end{equation}

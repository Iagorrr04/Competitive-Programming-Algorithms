\section{Bitwise}

\subsection{Binary to gray code}

$G_n = B_n$

$G_{n-1} = B_n \oplus B_{n-1} $

$G_{i} = B_i \oplus B_{i-1} $

$ ... $

$ G_1 = B_2 \oplus B_1 $

\subsection{Gray code to binary}


$ B_n = G_n $

$ B_{n-1} = B_n \oplus G_{n-1} = G_n \oplus G_{n-1} $

$ ... $

$ B_1 = B_2 \oplus G_1 = G_n \oplus G_1 $

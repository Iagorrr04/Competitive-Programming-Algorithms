\subsection{Inequalities}

\begin{equation}
        \begin{array}{c}
                x < a \\
                x < b \\
                x < min(a, b) \\
        \end{array}
\end{equation}

\begin{equation}
        \begin{array}{c}
                x <= a \\
                x <= b \\
                x <= min(a, b) \\
        \end{array}
\end{equation}

\begin{equation}
        \begin{array}{c}
                x > a \\
                x > b \\
                x > max(a, b) \\
        \end{array}
\end{equation}

\begin{equation}
        \begin{array}{c}
                x >= a \\
                x >= b \\
                x >= max(a, b) \\
        \end{array}
\end{equation}
